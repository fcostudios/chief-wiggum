### Step 1: Update `save_message` to accumulate session cost

In `src-tauri/src/commands/session.rs`, the `save_message` command calls `queries::insert_message()` but never updates session totals. Add the accumulation call:

```rust
#[tauri::command(rename_all = "snake_case")]
pub fn save_message(
    db: State<'_, Database>,
    session_id: String,
    id: String,
    role: String,
    content: String,
    model: Option<String>,
    input_tokens: Option<i64>,
    output_tokens: Option<i64>,
    cost_cents: Option<i64>,
) -> Result<(), AppError> {
    queries::insert_message(
        &db,
        &id,
        &session_id,
        &role,
        &content,
        model.as_deref(),
        input_tokens,
        output_tokens,
        cost_cents,
    )?;

    // Accumulate cost on the session row (CHI-53).
    // Only for roles that carry token/cost data (assistant, tool_result with errors).
    if input_tokens.is_some() || output_tokens.is_some() || cost_cents.is_some() {
        queries::update_session_cost(
            &db,
            &session_id,
            input_tokens.unwrap_or(0),
            output_tokens.unwrap_or(0),
            cost_cents.unwrap_or(0),
        )?;
    }

    Ok(())
}
```

/// Read CLAUDE.md from a project's folder path.
/// Returns the file content as a string, or an empty string if not found.
#[tauri::command(rename_all = "snake_case")]
pub fn read_claude_md(
    db: State<'_, Database>,
    project_id: String,
) -> Result<Option<String>, AppError> {
    let project = crate::db::queries::get_project(&db, &project_id)?
        .ok_or_else(|| AppError::Other(format!("Project {} not found", project_id)))?;

    let claude_md_path = std::path::Path::new(&project.path).join("CLAUDE.md");

    match std::fs::read_to_string(&claude_md_path) {
        Ok(content) => Ok(Some(content)),
        Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(None),
        Err(e) => Err(AppError::Other(format!("Failed to read CLAUDE.md: {}", e))),
    }
}
```

interface Command {
  id: string;
  label: string;
  category: string;
  shortcut?: string;
  icon?: () => JSX.Element;
  action: () => void;
}

function buildCommands(): Command[] {
  const commands: Command[] = [
    // Views
    {
      id: 'view-conversation',
      label: 'Switch to Conversation',
      category: 'Views',
      shortcut: '\u2318+1',
      icon: () => <MessageSquare size={14} />,
      action: () => setActiveView('conversation'),
    },
    {
      id: 'view-agents',
      label: 'Switch to Agents',
      category: 'Views',
      shortcut: '\u2318+2',
      icon: () => <Users size={14} />,
      action: () => setActiveView('agents'),
    },
    {
      id: 'view-diff',
      label: 'Switch to Diff',
      category: 'Views',
      shortcut: '\u2318+3',
      icon: () => <GitCompare size={14} />,
      action: () => setActiveView('diff'),
    },
    {
      id: 'view-terminal',
      label: 'Switch to Terminal',
      category: 'Views',
      shortcut: '\u2318+`',
      icon: () => <Terminal size={14} />,
      action: () => setActiveView('terminal'),
    },
    // Panels
    {
      id: 'toggle-sidebar',
      label: 'Toggle Sidebar',
      category: 'Panels',
      shortcut: '\u2318+B',
      icon: () => <PanelLeft size={14} />,
      action: () => toggleSidebar(),
    },
    {
      id: 'toggle-details',
      label: 'Toggle Details Panel',
      category: 'Panels',
      shortcut: '\u2318+\u21e7+B',
      icon: () => <PanelRight size={14} />,
      action: () => toggleDetailsPanel(),
    },
    // Session
    {
      id: 'new-session',
      label: 'New Session',
      category: 'Session',
      icon: () => <Plus size={14} />,
      action: () => {
        const project = getActiveProject();
        createNewSession(
          project?.default_model ?? 'claude-sonnet-4-6',
          project?.id,
        );
      },
    },
    {
      id: 'cycle-model',
      label: 'Cycle Model',
      category: 'Session',
      shortcut: '\u2318+M',
      icon: () => <Sparkles size={14} />,
      action: () => cycleModel(),
    },
  ];

  // Add session switchers dynamically
  for (const session of sessionState.sessions) {
    commands.push({
      id: `session-${session.id}`,
      label: session.title || 'Untitled Session',
      category: 'Sessions',
      icon: () => <MessageSquare size={14} />,
      action: () => {
        const oldId = sessionState.activeSessionId;
        setActiveSession(session.id);
        switchSession(session.id, oldId);
      },
    });
  }

  return commands;
}

/** Simple substring match scoring. Returns -1 for no match, or match position (lower is better). */
function fuzzyMatch(query: string, text: string): number {
  if (!query) return 0;
  const lowerQuery = query.toLowerCase();
  const lowerText = text.toLowerCase();
  const idx = lowerText.indexOf(lowerQuery);
  return idx;
}

export default function CommandPalette() {
  const [query, setQuery] = createSignal('');
  const [selectedIndex, setSelectedIndex] = createSignal(0);
  let inputRef: HTMLInputElement | undefined;
  let listRef: HTMLDivElement | undefined;

  const allCommands = () => buildCommands();

  const filteredCommands = () => {
    const q = query();
    if (!q) return allCommands();
    return allCommands()
      .filter((cmd) => fuzzyMatch(q, cmd.label) >= 0 || fuzzyMatch(q, cmd.category) >= 0)
      .sort((a, b) => fuzzyMatch(q, a.label) - fuzzyMatch(q, b.label));
  };

  function executeSelected() {
    const cmds = filteredCommands();
    const idx = selectedIndex();
    if (idx >= 0 && idx < cmds.length) {
      cmds[idx].action();
      closeCommandPalette();
    }
  }

  function handleKeyDown(e: KeyboardEvent) {
    if (e.key === 'Escape') {
      e.preventDefault();
      closeCommandPalette();
      return;
    }
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      setSelectedIndex((prev) => Math.min(prev + 1, filteredCommands().length - 1));
      scrollSelectedIntoView();
      return;
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault();
      setSelectedIndex((prev) => Math.max(prev - 1, 0));
      scrollSelectedIntoView();
      return;
    }
    if (e.key === 'Enter') {
      e.preventDefault();
      executeSelected();
      return;
    }
  }

  function scrollSelectedIntoView() {
    requestAnimationFrame(() => {
      const el = listRef?.querySelector('[data-selected="true"]');
      el?.scrollIntoView({ block: 'nearest' });
    });
  }

  function handleBackdropClick(e: MouseEvent) {
    if (e.target === e.currentTarget) {
      closeCommandPalette();
    }
  }

  onMount(() => {
    inputRef?.focus();
    document.addEventListener('keydown', handleKeyDown);
  });

  onCleanup(() => {
    document.removeEventListener('keydown', handleKeyDown);
  });

  // Reset selection when query changes
  const handleInput = (value: string) => {
    setQuery(value);
    setSelectedIndex(0);
  };

  // Group commands by category for display
  const groupedCommands = () => {
    const groups: Record<string, Command[]> = {};
    for (const cmd of filteredCommands()) {
      if (!groups[cmd.category]) groups[cmd.category] = [];
      groups[cmd.category].push(cmd);
    }
    return groups;
  };

  // Flat index tracker for keyboard nav across groups
  let flatIndex = -1;

  return (
    <div
      class="fixed inset-0 z-50 flex items-start justify-center pt-[20vh] bg-black/60"
      onClick={handleBackdropClick}
      role="dialog"
      aria-modal="true"
      aria-label="Command palette"
    >
      <div
        class="w-[560px] max-h-[400px] rounded-xl overflow-hidden animate-fade-in"
        style={{
          'background-color': 'var(--color-bg-primary)',
          border: '1px solid var(--color-border-primary)',
          'box-shadow': '0 25px 50px -12px rgba(0, 0, 0, 0.5)',
          'animation-duration': 'var(--duration-normal)',
        }}
      >
        {/* Search input */}
        <div
          class="flex items-center gap-3 px-4 py-3"
          style={{ 'border-bottom': '1px solid var(--color-border-secondary)' }}
        >
          <Search size={16} class="text-text-tertiary shrink-0" />
          <input
            ref={inputRef}
            type="text"
            placeholder="Type a command..."
            class="flex-1 bg-transparent text-sm text-text-primary outline-none placeholder:text-text-tertiary"
            value={query()}
            onInput={(e) => handleInput(e.currentTarget.value)}
          />
          <kbd class="text-[10px] text-text-tertiary px-1.5 py-0.5 rounded"
            style={{ 'background-color': 'var(--color-bg-tertiary)' }}
          >
            ESC
          </kbd>
        </div>

        {/* Results */}
        <div ref={listRef} class="overflow-y-auto max-h-[340px] py-1">
          <Show
            when={filteredCommands().length > 0}
            fallback={
              <div class="px-4 py-8 text-center text-sm text-text-tertiary">
                No commands found
              </div>
            }
          >
            {(() => {
              flatIndex = -1;
              return null;
            })()}
            <For each={Object.entries(groupedCommands())}>
              {([category, commands]) => (
                <>
                  <div class="px-3 py-1.5 text-[10px] font-semibold text-text-tertiary uppercase tracking-wider">
                    {category}
                  </div>
                  <For each={commands}>
                    {(cmd) => {
                      flatIndex++;
                      const myIndex = flatIndex;
                      const isSelected = () => selectedIndex() === myIndex;
                      return (
                        <button
                          class="flex items-center gap-3 w-full px-3 py-2 text-sm text-left transition-colors"
                          style={{
                            'background-color': isSelected()
                              ? 'var(--color-bg-tertiary)'
                              : 'transparent',
                            color: 'var(--color-text-primary)',
                          }}
                          data-selected={isSelected()}
                          onMouseEnter={() => setSelectedIndex(myIndex)}
                          onClick={() => {
                            cmd.action();
                            closeCommandPalette();
                          }}
                        >
                          <Show when={cmd.icon}>
                            <span class="text-text-tertiary shrink-0">{cmd.icon!()}</span>
                          </Show>
                          <span class="flex-1 truncate">{cmd.label}</span>
                          <Show when={cmd.shortcut}>
                            <kbd class="text-[10px] text-text-tertiary px-1.5 py-0.5 rounded shrink-0"
                              style={{ 'background-color': 'var(--color-bg-tertiary)' }}
                            >
                              {cmd.shortcut}
                            </kbd>
                          </Show>
                        </button>
                      );
                    }}
                  </For>
                </>
              )}
            </For>
          </Show>
        </div>
      </div>
    </div>
  );
}
```
